JOINS y Subconsultas

-- INNER JOIN
SELECT 
    p.nombre as producto,
    c.nombre as categoria,
    p.precio
FROM productos p
INNER JOIN categorias c 
    ON p.categoria_id = c.id;

-- LEFT JOIN con subconsulta
SELECT 
    u.nombre,
    u.email,
    (
        SELECT COUNT(*)
        FROM pedidos p
        WHERE p.usuario_id = u.id
    ) as total_pedidos
FROM usuarios u
LEFT JOIN pedidos p 
    ON u.id = p.usuario_id
WHERE u.activo = true;

-- Subconsulta en WHERE

FROM productos
WHERE precio > (
    SELECT AVG(precio)
    FROM productos

CREATE VIEW productos_disponibles AS
SELECT 
    p.id,
    p.nombre,
    p.precio,
    c.nombre as categoria
FROM productos p
JOIN categorias c ON p.categoria_id = c.id
WHERE p.stock > 0;

CREATE PROCEDURE crear_pedido(
    IN usuario_id INT,
    IN producto_id INT
)
BEGIN
    DECLARE stock_actual INT;
    
    -- Verificar stock
    SELECT stock INTO stock_actual
    FROM productos
    WHERE id = producto_id;
    
    IF stock_actual > 0 THEN
        -- Crear pedido
        INSERT INTO pedidos (usuario_id, producto_id)
        VALUES (usuario_id, producto_id);
        
        -- Actualizar stock
        UPDATE productos
        SET stock = stock - 1
        WHERE id = producto_id;
        
        SELECT 'Pedido creado' as mensaje;
    ELSE
        SELECT 'Sin stock' as mensaje;
    END IF;
END //
DELIMITER ;
